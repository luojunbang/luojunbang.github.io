//如果是组件 写清楚公用组件的用法 props 以及$emit 的方法
// <compA :text="text" :title="title" @confirm="" @change="" />
export default{
  componets:{},//组件
  mixins:[]//mixins, //mixins里面的方法命名为大写驼峰Cp.. || Ask... || Handle...
  //首先是属性
  props:{
    text:{
      type:[String,Number],
      required:true, //要么要求属性必须
      default:'--'  // 要么要有默认值
    }
  },
  created(){
    this._privateItem={} //不渲染在Dom的对象
  }
  data(){
    const _someStaticVal = {} //常量命名 每个常量应该命名
    return {
    someStaticVal:_someStaticVal,
    // shopItem:{} 直接会导致can get price of undefined
    shopItem:{
      price:0//由于有computed函数依赖，需初始化,对象设置默认属性的推荐写法
    },
    // Boolean
    isSome:false,
    canDo:false,
    hasSome:false,
    state:{},
    list:[]
  }}
  mounted(){},
  computed:{
    ...mapGetters(['getXXX']),vuex里面的getters统一 以get开头命名
    cpTotalPrice(){ //以cp开头命名
      let res = '0'
      if(options) res=this.shopItem.price*100 //依赖于shopItem的price
      return res
    }
  },
  methods:{
    ...mapMutations(['setXXX']),vuex里面的Mutations统一 以set开头命名
    // 函数单一原则
    handleDoSomething(){},//主动操作例如input，click事件等以handle开头
    onSomethingHappen(){},//被动操作 例如组件$emit,<CompA @click="onCompeClick" /> 监听dialog打开等
    onSomeCallback(){}, //不同操作的统一回调 以Callback结尾
    _resetPage(){},//页面私有方法 以_下划线开头
    _dataHandler(){},//页面私有方法 以_下划线开头
    //API
    askTotalState(){//API请求的统一以ask开头
      this.$API().then(res=>{
        Object.assign(this.state,res)
        // this.state = res 不建议这样
      })
    },
    askOrderList(){//API请求的统一以ask开头
      this.$API().then(res=>{
        this.list.splice(0,this.list.length) //请求列表建议先清空
        this.list.push(...res) //然后push进去
        // this.list = res 不建议这样
      })
    },
    askNewOrder(){//API请求的统一以ask开头
      // 表单请求
      let params = {}//请求参数处理 表单验证等等
      this.$API(params).then(res=>{})
    },

  },
  beforeDestroy(){
    // 页面有timer等需要清除
    // clearTimeOut()
    // clearInterval()
  },
}
