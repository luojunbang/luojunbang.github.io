# Money 类Handler

const _isPrice = function (num) {
    let regPos = /^(0|[1-9]\d*)(\.\d+)?$/
    // let regNeg = /^(-(([0-9]+\.[0-9]*[1-9][0-9]*)|([0-9]*[1-9][0-9]*\.[0-9]+)|([0-9]*[1-9][0-9]*)))$/; //负浮点数
    // || regNeg.test(num)
    if (regPos.test(num)) {
        return true
    } else {
        return false
    }
}
export const isPrice = _isPrice

/**
 * @desc 生成规范金额字符串
 * @param s | String | Number
 * @param bool | Boolean 是否携带¥
 * @returns String
 */
export function fmtMoney(s, bool = true) {
    if (/[^0-9.]/.test(s)) return (bool ? '¥ ' : '') + '0.00'
    if (s == null || s == 'null' || s == '') return '0.00'
    s = (s / 100).toFixed(2)
    s = s.toString().replace(/^(\d*)$/, '$1.')
    s = (s + '00').replace(/(\d*\.\d\d)\d*/, '$1')
    s = s.replace('.', ',')
    let re = /(\d)(\d{3},)/
    while (re.test(s)) s = s.replace(re, '$1,$2')
    s = s.replace(/,(\d\d)$/, '.$1')
    return (bool ? '¥ ' : '') + s
}

/**
 * @desc 生成大写金额字符串
 * @param n | String | Number
 * @param isYuan | Boolean 是否为元为单位 false则表示分
 * @returns 写大金额字符串
 */

export function fmtMoneyCN(n, isYuan = true) {
    if (!_isPrice(n)) return '数据非法'
    if (!isYuan) n = n * 100
    n = Math.floor(n).toFixed(0)
    const UNIT = '仟佰拾亿仟佰拾萬仟佰拾元角分',
        CN_NUM = '零壹贰叁肆伍陆柒捌玖'
    let unit = UNIT.substr(UNIT.length - n.length),
        str = Array.from(n).reduce((res, num, idx) => (res += CN_NUM.charAt(num) + unit.charAt(idx)), '')
    return str
        .replace(/零(仟|佰|拾|角)/g, '零')
        .replace(/(零)+/g, '零')
        .replace(/零(萬|亿|元)/g, '$1')
        .replace(/(亿)萬|壹(拾)/g, '$1$2')
        .replace(/^元零?|零分/g, '')
        .replace(/元$/g, '元整')
}
