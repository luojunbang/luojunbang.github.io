# URL 类 Handler

# 获取 url 的参数

/**
 * @param {string} url [luojunbang.github.io?key=val]
 * @return {string} obj [{key:val}]
 */
export function getParamsBy(url = "") {
  let obj = {};
  const search = url.substring(url.lastIndexOf("?") + 1);
  const reg = /([^?&=]+)=([^?&=]*)/g; //不匹配&和=
  search.replace(reg, (rs, $1, $2) => {
    const key = decodeURIComponent($1);
    const val = decodeURIComponent($2);
    val = String(val);
    obj[key] = val;
    return rs;
  });
  return obj;
}

# 根据参数组成 url

/**
 * @param {obj}  [key:val]
 * @param {url}  [luojunbang.github.io]
 * @return {string} obj [luojunbang.github.io?key=val]
 */
export function parseUrl(params, url) {
  let res = url && url.indexOf("?") > -1 ? url : url + "?";
  res = res === "?" ? "" : res;
  return (
    res +
    Object.keys(params).reduce((res, key) => {
      return res += `${encodeURIComponent(key)}=${encodeURIComponent(params[key])}&`;
    }, "")
  );
}
