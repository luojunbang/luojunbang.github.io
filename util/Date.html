# DATE类 Handler

const fmt2 = function (num) {
    return num < 10 ? '0' + num : num
}

/**
 * @desc 生成日期对象
 * @param Date | String | Number
 * @returns //2020-01-01
 */
const generatorDate = function (date) {
    let d = null
    if (!date) return d
    if (/^[0-9]{0,13}$/.test(date)) {
        d = new Date(Math.floor(date))
    } else {
        if (Object.prototype.toString.call(date) === '[object Date]') d = date
        else d = new Date(date)
    }
    if (d.toString() === 'Invalid Date') return null
    return d
}

/**
 * @param Date | String | Number
 * @returns //2020-01-01
 */
const _fmtDate = function (date, splitor = '-') {
    const d = generatorDate(date)
    if (!d) return 'err_Date'
    return `${d.getFullYear()}${splitor}${fmt2(d.getMonth() + 1)}${splitor}${fmt2(d.getDate())}`
}
/**
 * @param Date | String | Number
 * @returns //18:00:00
 */
const _fmtTime = function (date, splitor = ':') {
    const d = generatorDate(date)
    if (!d) return 'err_Date'
    return `${d.getHours()}${splitor}${fmt2(d.getMinutes() + 1)}${splitor}${fmt2(d.getSeconds())}`
}
/**
 * @param Date | String | Number
 * @returns 2020-01-01 18:00:00
 */

const _fmtDateTime = function (date) {
    const d = generatorDate(date)
    if (!d) return 'err_Date'
    return `${d.getFullYear()}-${fmt2(d.getMonth() + 1)}-${fmt2(d.getDate())} ${d.getHours()}:${fmt2(d.getMinutes() + 1)}:${fmt2(d.getSeconds())}`
}

const _isSecondTimeBigger = function (start, end) {
    return generatorDate(end) > generatorDate(start)
}

export const fmtDate = _fmtDate
export const fmtTime = _fmtTime
export const fmtDateTime = _fmtDateTime
export const isSecondTimeBigger = _isSecondTimeBigger
